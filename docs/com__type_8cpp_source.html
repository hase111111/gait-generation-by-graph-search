116bool IsLegPairFree(int leg_index, int leg_ground_pattern_index)
117{
118 LegGroundedBit leg_ground_pattern;
119
120 // indexから遊脚のパターンを取得する．
121 try
122 {
123 leg_ground_pattern =
124 ::kLegGroundedPatternMap.right.at(leg_ground_pattern_index);
125 }
126 catch (...)
127 {
128 return false;
129 }
130
131 // 両隣が遊脚の場合は true を返す．
132 if (!leg_ground_pattern[leg_index % HexapodConst::kLegNum] &&
133 !leg_ground_pattern[(leg_index + 1) % HexapodConst::kLegNum])
134 {
135 return true;
136 }
137 else
138 {
139 return false;
140 }
141}

145std::unordered_map<DiscreteComPos, std::vector<int>> MakeLegGroundedPatternBanList()
146{
147 std::unordered_map<DiscreteComPos, std::vector<int>> res;
148
149
150 // ロボットの体が前に寄っている時に前足が両方とも遊脚だと転倒してしまう．
151 // そのため，離散化された重心から，とることができない，
152 // 連続する脚が両方とも遊脚になるパターンを禁止するのがこの関数の目的である．
153 std::unordered_map<DiscreteComPos, std::vector<int>> ban_leg_index_list;
154 ban_leg_index_list[DiscreteComPos::kFront] = { 0, 4, 5 };
155 ban_leg_index_list[DiscreteComPos::kFrontRight] = { 0, 1, 5 };
156 ban_leg_index_list[DiscreteComPos::kFrontLeft] = { 3, 4, 5 };
157 ban_leg_index_list[DiscreteComPos::kBack] = { 1, 2, 3 };
158 ban_leg_index_list[DiscreteComPos::kBackRight] = { 0, 1, 2 };
159 ban_leg_index_list[DiscreteComPos::kBackLeft] = { 2, 3, 4 };
160 ban_leg_index_list[DiscreteComPos::kCenterBack] = { 0, 2, 4 };
161 ban_leg_index_list[DiscreteComPos::kCenterFront] = { 1, 3, 5 };
162
163
164 // DiscreteComPosの要素数だけループする．
165 // magic_enum::enum_values<DiscreteComPos>() は，
166 // DiscreteComPos の要素を列挙した array を返す．
167 for (const auto& i : magic_enum::enum_values<DiscreteComPos>())
168 {
169 if (ban_leg_index_list.count(i) == 0) { continue; }
170
171 for (auto& j : ban_leg_index_list[i])
172 {
173 for (int k = 0; k < ::kLegGroundedPatternNum; ++k)
174 {
175 if (IsLegPairFree(j, k))
176 {
177 res[i].push_back(k);
178 }
179 }
180 }
181 }
182
183 return std::move(res);
184}

188std::vector<std::vector<int>> MakeLegGroundedPatternBanListFromLeg()
189{
190 std::vector<std::vector<int>> res;
191
192 res.resize(designlab::HexapodConst::kLegNum); // 脚の数だけ vector を確保する．
193
194 // i 番脚を接地しなければ，取ることができないものを保存する．
195 for (int i = 0; i < designlab::HexapodConst::kLegNum; i++)
196 {
197 for (int j = 0; j < ::kLegGroundedPatternNum; ++j)
198 {
199 // i番目のビットを確認し，立っているならば
200 // (つまり，その脚を接地しなければいけないなら)，そのパターンを禁止する．
201 if (::kLegGroundedPatternMap.right.at(j)[i])
202 {
203 res[i].push_back(j);
204 }
205 }
206 }
207
208 return std::move(res);
209}
