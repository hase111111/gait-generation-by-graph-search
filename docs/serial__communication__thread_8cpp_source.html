29void SerialCommunicationThread::Loop()
30{
31 if (!Initialize())
32 {
33 end_flag_ = true;
34
35 // 通信の初期化に失敗した場合は，即座に終了する．
36 return;
37 }
38
39 while (true)
40 {
41 Sleep(static_cast<int>(1000 * kThreadPeriod)); // 指定した時間だけスリープする．(ミリ秒単位)
42
43 // 読み込みを行う．
44 if (!Read())
45 {
46 EndThread();
47
48 // 読み込みに失敗した場合は，即座に終了する．
49 return;
50 }
51
52 // 書き込みを行う．
53 if (!Write())
54 {
55 EndThread();
56
57 // 書き込みに失敗した場合は，即座に終了する．
58 return;
59 }
60
61 // スレッドの終了フラグが立っている場合は，即座に終了する．
62 if (IsEnd())
63 {
64 return;
65 }
66 }
67}

105std::vector<std::string> SerialCommunicationThread::GetReadData(const int num) const
106{
107 // スレッドの排他制御を行う．
108 boost::mutex::scoped_lock lock(mutex_);
109
110 std::vector<std::string> ret;
111
112 // 読み込みデータが空の場合は，空の配列を返す．
113 if (read_data_.empty())
114 {
115 return ret;
116 }
117
118 // 読み込みデータが指定した数よりも少ない場合は，全てのデータを返す．
119 if (read_data_.size() <= num)
120 {
121 return read_data_;
122 }
123
124 // 読み込みデータが指定した数よりも多い場合は，最新のデータを返す．
125 for (int i = 0; i < num; ++i)
126 {
127 ret.push_back(read_data_[read_data_.size() - 1 - i]);
128 }
129
130 return ret;
131}

133bool SerialCommunicationThread::Initialize()
134{
135 // 通信を開始する．Windowsでは CreateFile で COMポートを開く．
136 serial_handle_ = CreateFile(kComName[7].c_str(), GENERIC_READ | GENERIC_WRITE,
137 0, NULL, OPEN_EXISTING, FILE_ATTRIBUTE_NORMAL, NULL);
138
139 if (serial_handle_ == INVALID_HANDLE_VALUE)
140 {
141 return false;
142 }
143
144 // 通信の設定を行う．
145 SetupComm(serial_handle_, kBufferSize, kBufferSize); // バッファーを作る．
146 PurgeComm(serial_handle_, PURGE_TXABORT | PURGE_RXABORT | PURGE_TXCLEAR | PURGE_RXCLEAR); // バッファーのクリア．
147
148 // シリアルポートの設定を行う．
149 DCB dcb;
150 GetCommState(serial_handle_, &dcb); // 現在の設定を取得．
151
152 dcb.ByteSize = 8;
153 dcb.XonChar = 0;
154 dcb.XoffChar = 0;
155 dcb.EofChar = 4;
156 dcb.EvtChar = 10;
157 dcb.BaudRate = kBaudRate;
158
159 SetCommState(serial_handle_, &dcb); // 設定を反映．
160
161 // タイムアウトの設定を行う．
162 COMMTIMEOUTS timeouts;
163 GetCommTimeouts(serial_handle_, &timeouts); // 現在の設定を取得．
164
165 timeouts.ReadIntervalTimeout = 100;
166 timeouts.ReadTotalTimeoutMultiplier = 100;
167 timeouts.ReadTotalTimeoutConstant = 100;
168 timeouts.WriteTotalTimeoutMultiplier = 0;
169 timeouts.WriteTotalTimeoutConstant = 0;
170
171 SetCommTimeouts(serial_handle_, &timeouts); // 設定を反映．
172
173 return true;
174}

176bool SerialCommunicationThread::Read()
177{
178 // 通信が開始されていない場合は，即座に終了する．
179 if (serial_handle_ == INVALID_HANDLE_VALUE)
180 {
181 return false;
182 }
183
184 DWORD read_size = 0;
185 char read_buffer[kBufferSize] = { 0 };
186
187 if (!ReadFile(serial_handle_, read_buffer, kBufferSize, &read_size, NULL))
188 {
189 // 読み込みに失敗した場合は終了する．
190 return false;
191 }
192
193 if (read_size == 0)
194 {
195 // 読み込みデータがない場合は終了する．
196 // 読み込み自体は成功しているため，trueを返す．
197 return true;
198 }
199
200 // 読み込んだデータを格納する．
201 std::string read_data(read_buffer, read_size);
202
203 // 読み込んだデータを改行コードで分割する．
204 std::vector<std::string> split_data = string_util::Split(read_data, "\n");
205
206 // 分割したデータを読み込みデータに追加する．
207 {
208 // スレッドの排他制御を行う．
209 boost::mutex::scoped_lock lock(mutex_);
210
211 for (const auto& data : split_data)
212 {
213 // 空文字の場合は追加しない．
214 if (data.empty())
215 {
216 continue;
217 }
218
219 // 読み込みデータに追加する．
220 read_data_.push_back(data);
221 }
222 }
223
224 return true;
225}

227bool SerialCommunicationThread::Write()
228{
229 // 通信が開始されていない場合は，即座に終了する．
230 if (serial_handle_ == INVALID_HANDLE_VALUE)
231 {
232 return false;
233 }
234
235 // 書き込みデータがない場合は，即座に終了する．
236 if (write_data_ == "")
237 {
238 return true;
239 }
240
241 DWORD write_size = 0;
242
243 if (!WriteFile(serial_handle_, write_data_.c_str(), static_cast<DWORD>(write_data_.size()), &write_size, NULL))
244 {
245 // 書き込みに失敗した場合は終了する．
246 return false;
247 }
248
249 if (write_size == 0)
250 {
251 // 書き込みデータがない場合は終了する．
252 // 書き込み自体は成功しているため，trueを返す．
253 return true;
254 }
255
256 // 書き込みデータをクリアする．
257 write_data_ = "";
258
259 return true;
260}
