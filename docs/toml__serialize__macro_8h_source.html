401#define DESIGNLAB_TOML11_SERIALIZE(NAME, ...) \
402namespace toml \
403{ \
404template<> \
405struct from<NAME> \
406{ \
407 static_assert(std::is_class<NAME>::value, \
408 "第1引数はクラスか構造体である必要があります．"); \
409 static_assert(std::is_default_constructible<NAME>::value, \
410 "第1引数はデフォルトコンストラクタを持つ必要があります．"); \
411 \
412 template<typename C, template<typename ...> class T, \
413 template<typename ...> class A> \
414 static NAME from_toml(basic_value<C, T, A>& v) \
415 { \
416 ::toml::basic_value<toml::preserve_comments, std::map> v_ = v; \
417 NAME obj; \
418 NAME##Description desc; \
419 TOML11_FOR_EACH_VA_ARGS( \
420 DESIGNLAB_SUB_MACRO_FIND_MEMBER_VARIABLE_FROM_VALUE, __VA_ARGS__) \
421 return obj; \
422 } \
423}; \
424 \
425template<> \
426struct into<NAME> \
427{ \
428 static value into_toml(const NAME& obj) \
429 { \
430 ::toml::basic_value<toml::preserve_comments, std::map> v = ::toml::table{}; \
431 \
432 NAME##Description desc; \
433 \
434 for (const auto i : desc.file_description_vec) \
435 { \
436 v.comments().push_back(i); \
437 } \
438 \
439 for (int i = 0; i < desc.table_name_description_vec.size(); ++i) \
440 { \
441 v[desc.table_name_description_vec[i]] = ::toml::table{}; \
442 v[desc.table_name_description_vec[i]].comments(). \
443 push_back(desc.table_name_description_vec[i + 1]); \
444 ++i; \
445 } \
446 \
447 TOML11_FOR_EACH_VA_ARGS( \
448 DESIGNLAB_SUB_MACRO_ASSIGN_MEMBER_VARIABLE_TO_VALUE, __VA_ARGS__) \
449 TOML11_FOR_EACH_VA_ARGS( \
450 DESIGNLAB_SUB_MACRO_ADD_COMMENT, __VA_ARGS__) \
451 return v; \
452 } \
453}; \
