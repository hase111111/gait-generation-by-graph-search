24void SystemMainRobotControl::Main()
25{
26 using enum OutputDetail;
27
28 CmdIOUtil::OutputTitle("Result Viewer System");
29
30 while (true)
31 {
32 // ファイルツリーを表示し，ファイルを選択する．
33 FileTree file_tree;
34
35 std::string res_path;
36
37 if (!file_tree.SelectFile(kResultFileDirectoryPath, -1, "csv", ResultFileConst::kNodeListName, &res_path))
38 {
39 CmdIOUtil::Output("該当のデータがありませんでした．終了します．", kSystem);
40
41 break;
42 }
43
44 // ファイルを読み込む．
45
46 std::vector<RobotStateNode> graph; // データを受け取るための変数．
47 MapState map_state;
48
49 if (result_importer_.ImportNodeListAndMapState(res_path, &graph, &map_state))
50 {
51 RemoveDoNotMoveNode(&graph);
52 MergeContinuousMove(&graph);
53 DivideSwingAndStance(&graph);
54
55 // データを仲介人に渡す．
56 broker_ptr_->graph.SetData(graph);
57 broker_ptr_->map_state.SetData(map_state);
58 broker_ptr_->simulation_end_index.SetData({ graph.size() - 1 });
59
60 // データを表示する．
61 CmdIOUtil::Output("データを表示します．", kSystem);
62 CmdIOUtil::OutputNewLine(1, kSystem);
63 CmdIOUtil::WaitAnyKey();
64 CmdIOUtil::OutputNewLine(1, kSystem);
65 CmdIOUtil::OutputHorizontalLine("=", kSystem);
66 }
67 else
68 {
69 CmdIOUtil::Output("ファイルの読み込みに失敗しました．終了します．", kSystem);
70 }
71
72 // 終了するかどうかを選択
73
74 if (CmdIOUtil::InputYesNo("このモードを終了しますか？"))
75 {
76 CmdIOUtil::OutputNewLine(1, kSystem);
77
78 break;
79 }
80
81 CmdIOUtil::OutputNewLine(1, kSystem);
82 }
83};

85void SystemMainRobotControl::RemoveDoNotMoveNode(std::vector<RobotStateNode>* graph_ptr)
86{
87 // 脚先の座標のみを確認し，変更がない場合は削除する．
88 std::optional<RobotStateNode> prev_pos = std::nullopt;
89
90 for (auto itr = graph_ptr->begin(); itr != graph_ptr->end();)
91 {
92 if (prev_pos.has_value() && itr->leg_pos == prev_pos->leg_pos)
93 {
94 itr = graph_ptr->erase(itr);
95 }
96 else
97 {
98 prev_pos = *itr;
99 ++itr;
100 }

104void SystemMainRobotControl::MergeContinuousMove(std::vector<RobotStateNode>* graph_ptr)
105{
106 // 胴体の平行移動を連続して行うノードをまとめる．
107 std::optional<RobotStateNode> prev_pos = std::nullopt;
108
109 bool prv_is_body_move = false;
110
111 for (auto itr = graph_ptr->begin(); itr != graph_ptr->end();)
112 {
113 if (prev_pos.has_value() &&
114 (*itr).center_of_mass_global_coord != prev_pos.value().center_of_mass_global_coord &&
115 (*itr).center_of_mass_global_coord.z == prev_pos.value().center_of_mass_global_coord.z)
116 {
117 prev_pos = *itr;
118
119 if (prv_is_body_move)
120 {
121 (*(itr - 1)) = *itr;
122 itr = graph_ptr->erase(itr);
123 prv_is_body_move = false;
124 }
125 else
126 {
127 prv_is_body_move = true;
128 ++itr;
129 }
130 }
131 else
132 {
133 prv_is_body_move = false;
134 prev_pos = *itr;
135 ++itr;
136 }
137 }

140void SystemMainRobotControl::DivideSwingAndStance(std::vector<RobotStateNode>* graph_ptr)
141{
142 // 脚先の座標のみを確認し，変更がない場合は削除する．
143 std::optional<RobotStateNode> prev_pos = std::nullopt;
144
145 for (auto itr = graph_ptr->begin(); itr != graph_ptr->end(); ++itr)
146 {
147 // 前回のノードがないならば，更新して次へ．
148 if (!prev_pos.has_value())
149 {
150 prev_pos = *itr;
151 continue;
152 }
153
154 // 脚先の座標のZ座標，つまり高さが変化しているかどうかを確認する．
155 int swing_leg_num = 0;
156 int stance_leg_num = 0;
157
158 for (int i = 0; i < HexapodConst::kLegNum; i++)
159 {
160 if (!leg_func::IsGrounded(prev_pos.value().leg_state, i) && leg_func::IsGrounded((*itr).leg_state, i))
161 {
162 ++stance_leg_num;
163 }
164
165 if (leg_func::IsGrounded(prev_pos.value().leg_state, i) && !leg_func::IsGrounded((*itr).leg_state, i))
166 {
167 ++swing_leg_num;
168 }
169 }
170
171 // 遊脚・接地を分ける．
172 if (swing_leg_num > 0 && stance_leg_num > 0)
173 {
174 RobotStateNode stance_node = prev_pos.value();
175
176 for (int i = 0; i < HexapodConst::kLegNum; i++)
177 {
178 // 前に遊脚，現在支持脚の場合のみ処理をする．
179 if (!leg_func::IsGrounded(prev_pos.value().leg_state, i) && leg_func::IsGrounded((*itr).leg_state, i))
180 {
181 // 脚位置の変更
182 stance_node.leg_pos[i] = (*itr).leg_pos[i];
183
184 // 脚の状態の変更
185 leg_func::ChangeGround(i, true, &stance_node.leg_state);
186 }
187 }
188
189 // ノードを挿入する．
190 itr = graph_ptr->insert(itr, stance_node);
191 }
192
193 prev_pos = *itr;
194 }
195}
