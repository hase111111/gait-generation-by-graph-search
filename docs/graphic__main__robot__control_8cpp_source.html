88bool GraphicMainRobotControl::Update()
89{
90 assert(mouse_ptr_ != nullptr);
91
92 mouse_ptr_->Update();
93
94 gui_updater_.Activate(mouse_ptr_);
95
96 if (!graph_.empty())
97 {
98 // 表示するノードを取得する．
99 size_t display_node_index_ = display_node_switch_gui_->GetDisplayNodeNum();
100
101 // ノードの情報を設定するGUIに情報を伝達する．
102 node_setter_group_.SetNode(graph_.at(display_node_index_));
103 }
104
105 if (broker_->graph.GetUpdateCount() != graph_update_count_)
106 {
107 graph_update_count_ = broker_->graph.GetUpdateCount();
108
109 graph_ = broker_->graph.GetData();
110
111 display_node_switch_gui_->SetGraphData(graph_.size(), { graph_.size() - 1 });
112 }
113
114 if (broker_->map_state.GetUpdateCount() != map_update_count_)
115 {
116 map_update_count_ = broker_->map_state.GetUpdateCount();
117
118 map_state_ = broker_->map_state.GetData();
119
120 divided_map_state_.Init(map_state_, {});
121
122 map_renderer_->SetMapState(map_state_);
123 }
124
125 return true;
126}
