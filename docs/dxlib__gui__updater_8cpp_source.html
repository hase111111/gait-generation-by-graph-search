19void DxlibGuiUpdater::OpenTerminal()
20{
21 // 2回以上呼ばれたら何もしない．
22 if (is_terminal_opened_)
23 {
24 return;
25 }
26
27 // ターミナルに渡すGUIのリストを作成．
28 std::vector<std::shared_ptr<IDxlibGui> > gui_list;
29
30 for (const auto& i : gui_ptrs_)
31 {
32 gui_list.push_back(i.second);
33 }
34
35 // ターミナルを登録．
36 const auto terminal_ptr = std::make_shared<DxlibGuiTerminal>(gui_list);
37 const int terminal_priority = kTopPriority + 1;
38
39 RegisterGui(terminal_ptr, terminal_priority);
40 RegisterClickable(terminal_ptr, terminal_priority);
41
42 is_terminal_opened_ = true;
43}

68void DxlibGuiUpdater::RegisterGui(
69 const std::shared_ptr<IDxlibGui> gui_ptr, int priority)
70{
71 assert(gui_ptr != nullptr);
72
73 // すでに同じポインタが登録されていたら何もしない．
74 for (const auto& i : gui_ptrs_)
75 {
76 if (i.second == gui_ptr)
77 {
78 return;
79 }
80 }
81
82 // すでに同じ優先度のものがあったら，orderを1つずつずらす．
83 Priority p{ priority, 0 };
84
85 while (gui_ptrs_.find(p) != gui_ptrs_.end())
86 {
87 p.order++;
88 }
89
90 // 登録．
91 gui_ptrs_[p] = gui_ptr;

181void DxlibGuiUpdater::ActivateClickable(const std::shared_ptr<const Mouse> mouse_ptr)
182{
183 // 優先度の高いものから順にクリック判定を行う．
184 // 昇順(増える順)に並んでいるので，rbegin から rend まで(降順に)走査する．
185
186 for (auto i = clickable_ptrs_.rbegin(); i != clickable_ptrs_.rend(); ++i)
187 {
188 if ((*i).second->CursorOnGui(
189 mouse_ptr->GetCursorPosX(),
190 mouse_ptr->GetCursorPosY()))
191 {
192 (*i).second->ClickedAction({
193 .cursor_x = mouse_ptr->GetCursorPosX(),
194 .cursor_y = mouse_ptr->GetCursorPosY(),
195 .left_pushing_count =
196 mouse_ptr->GetPressingCount(MOUSE_INPUT_LEFT),
197 .middle_pushing_count =
198 mouse_ptr->GetReleasingCount(MOUSE_INPUT_MIDDLE),
199 .right_pushing_count =
200 mouse_ptr->GetPressingCount(MOUSE_INPUT_RIGHT) });
201
202 break;
203 }
204 }

207void DxlibGuiUpdater::ActivateDraggable(const std::shared_ptr<const Mouse> mouse_ptr)
208{
209 unsigned int mouse_key{ 0 };
210 int pressing_count{ 0 };
211 std::array<unsigned int, 3> mouse_keys = {
212 MOUSE_INPUT_LEFT, MOUSE_INPUT_MIDDLE, MOUSE_INPUT_RIGHT };
213
214 for (auto i : mouse_keys)
215 {
216 if (mouse_ptr->GetPressingCount(i) > 0)
217 {
218 mouse_key += i;
219 pressing_count =
220 (std::max)(mouse_ptr->GetPressingCount(i), pressing_count);
221 }
222 }
223
224
225 if (pressing_count == 0)
226 {
227 // 左クリックが押されていないならば全てのドラッグを終了する
228 for (auto& i : draggable_ptrs_)
229 {
230 i.second->SetDragged(false);
231 }
232
233 now_dragging_gui_key_ = std::nullopt;
234 }
235 else if (pressing_count == 1)
236 {
237 // 左クリックが押された瞬間，ドラッグを開始する
238 for (auto i = draggable_ptrs_.rbegin(); i != draggable_ptrs_.rend(); ++i)
239 {
240 if ((*i).second->IsDraggable(
241 mouse_ptr->GetCursorPosX(),
242 mouse_ptr->GetCursorPosY()))
243 {
244 (*i).second->SetDragged(true);
245 now_dragging_gui_key_ = (*i).first;
246 break;
247 }
248 }
249 }
250 else if (pressing_count > 0)
251 {
252 // 左クリックが押され続けているならばドラッグ判定を行う．
253 if (now_dragging_gui_key_.has_value() &&
254 draggable_ptrs_.count(now_dragging_gui_key_.value()) == 1)
255 {
256 draggable_ptrs_[now_dragging_gui_key_.value()]->DraggedAction(
257 mouse_ptr->GetDiffPosX(), mouse_ptr->GetDiffPosY(), mouse_key);
258 }
259 }
260 else
261 {
262 assert(false); // 冗長
263 }
264}

266void DxlibGuiUpdater::ActivateWheelHandler(
267 const std::shared_ptr<const Mouse> mouse_ptr)
268{
269 if (mouse_ptr->GetWheelRot() == 0) { return; }
270
271 // 優先度の高いものから順にホイール操作判定を行う
272 // 昇順(増える順)に並んでいるので，rbeginからrendまで(降順に)走査する
273
274 for (auto i = wheel_handler_ptrs_.rbegin(); i != wheel_handler_ptrs_.rend(); ++i)
275 {
276 if ((*i).second->CanHandleWheel(
277 mouse_ptr->GetCursorPosX(),
278 mouse_ptr->GetCursorPosY()))
279 {
280 (*i).second->RotMouseWheel(mouse_ptr->GetWheelRot());
281 break;
282 }
283 }
284}
