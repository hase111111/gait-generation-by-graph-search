331void DxlibGuiNodeDisplayer::DrawJointInfo() const
332{
333 using math_util::FloatingPointNumToString;
334 using math_util::ConvertRadToDeg;
335
336 const unsigned int text_color = GetColor(10, 10, 10);
337 const unsigned int error_text_color = GetColor(128, 10, 10);
338 const int text_pos_x = gui_left_pos_x_ + 10;
339 const int text_pos_y_min = gui_top_pos_y_ + kTitleBarHeight + 10;
340 const int text_interval_y = 20;
341
342 if (!calculator_ptr_)
343 {
344 DrawFormatStringToHandle(
345 text_pos_x,
346 text_pos_y_min + text_interval_y * 0,
347 text_color,
348 font_handle_,
349 "計算クラスが nullptr です");
350 return;
351 }
352 if (!checker_ptr_)
353 {
354 DrawFormatStringToHandle(
355 text_pos_x,
356 text_pos_y_min + text_interval_y * 0,
357 text_color,
358 font_handle_,
359 "チェッカークラスが nullptr です");
360 return;
361 }
362
363 for (int i = 0; i < HexapodConst::kLegNum; i++)
364 {
365 if (joint_state_[i].joint_angle.size() != 3)
366 {
367 DrawFormatStringToHandle(
368 text_pos_x,
369 text_pos_y_min + text_interval_y * 0,
370 text_color,
371 font_handle_,
372 "間接の計算ができていない，またはされていません．");
373
374 return;
375 }
376 if (joint_state_[i].joint_pos_leg_coordinate.size() != 4)
377 {
378 DrawFormatStringToHandle(
379 text_pos_x,
380 text_pos_y_min + text_interval_y * 0,
381 text_color,
382 font_handle_,
383 "間接の計算ができていない，またはされていません．");
384
385 return;
386 }
387 }
388
389 int text_line = 0;
390
391
392 for (int i = 0; i < HexapodConst::kLegNum; i++)
393 {
394 const float coxa_angle_deg =
395 ConvertRadToDeg(joint_state_[i].joint_angle[0]);
396 const float femur_angle_deg =
397 ConvertRadToDeg(joint_state_[i].joint_angle[1]);
398 const float tibia_angle_deg =
399 ConvertRadToDeg(joint_state_[i].joint_angle[2]);
400
401 DrawFormatStringToHandle(
402 text_pos_x,
403 text_pos_y_min + text_interval_y * (text_line++),
404 text_color,
405 font_handle_,
406 "[%d] c %s[deg],f %s[deg],t %s[deg]",
407 i,
408 FloatingPointNumToString(coxa_angle_deg).c_str(),
409 FloatingPointNumToString(femur_angle_deg).c_str(),
410 FloatingPointNumToString(tibia_angle_deg).c_str());
411
412 const auto coxa_to_femur = joint_state_[i].joint_pos_leg_coordinate[0] -
413 joint_state_[i].joint_pos_leg_coordinate[1];
414 const auto femur_to_tibia = joint_state_[i].joint_pos_leg_coordinate[1] -
415 joint_state_[i].joint_pos_leg_coordinate[2];
416 const auto tibia_to_end = joint_state_[i].joint_pos_leg_coordinate[2] -
417 joint_state_[i].joint_pos_leg_coordinate[3];
418
419 DrawFormatStringToHandle(
420 text_pos_x,
421 text_pos_y_min + text_interval_y * (text_line++),
422 text_color,
423 font_handle_,
424 " c %3.3f[mm],f %3.3f[mm],t %3.3f[mm]",
425 coxa_to_femur.GetLength(),
426 femur_to_tibia.GetLength(),
427 tibia_to_end.GetLength());
428
429
430 if (checker_ptr_->IsLegInRange(
431 i, joint_state_[i].joint_pos_leg_coordinate[3]))
432 {
433 DrawFormatStringToHandle(
434 text_pos_x,
435 text_pos_y_min + text_interval_y * (text_line++),
436 text_color,
437 font_handle_,
438 " 近似値された可動域内にあります．");
439 }
440 else
441 {
442 DrawFormatStringToHandle(
443 text_pos_x,
444 text_pos_y_min + text_interval_y * (text_line++),
445 error_text_color,
446 font_handle_,
447 " 近似値された可動域外です．");
448 }
449
450
451 std::string str = "";
452
453 const float coxa_min_angle =
454 PhantomXMkIIConst::kCoxaDefaultAngle[i] +
455 PhantomXMkIIConst::kCoxaAngleMin;
456 const float coxa_max_angle =
457 PhantomXMkIIConst::kCoxaDefaultAngle[i] +
458 PhantomXMkIIConst::kCoxaAngleMax;
459
460 if (joint_state_[i].joint_angle[0] < coxa_min_angle)
461 {
462 str += "coxa_min ";
463 }
464
465 if (joint_state_[i].joint_angle[0] > coxa_max_angle)
466 {
467 str += "coxa_max ";
468 }
469
470 if (joint_state_[i].joint_angle[1] < PhantomXMkIIConst::kFemurAngleMin)
471 {
472 str += "femur_min ";
473 }
474
475 if (joint_state_[i].joint_angle[1] > PhantomXMkIIConst::kFemurAngleMax)
476 {
477 str += "femur_max ";
478 }
479
480 if (joint_state_[i].joint_angle[2] < PhantomXMkIIConst::kTibiaAngleMin)
481 {
482 str += "tibia_min ";
483 }
484
485 if (joint_state_[i].joint_angle[2] > PhantomXMkIIConst::kTibiaAngleMax)
486 {
487 str += "tibia_max ";
488 }
489
490 if (!joint_state_[i].is_in_range)
491 {
492 str += "脚先が届いていません ";
493 }
494
495 if (!str.empty())
496 {
497 const size_t max_str_size = 30;
498 if (str.size() > max_str_size) { str = str.substr(0, max_str_size); }
499
500 DrawFormatStringToHandle(
501 text_pos_x,
502 text_pos_y_min + text_interval_y * (text_line++),
503 error_text_color,
504 font_handle_,
505 " 実際の可動域の外です． %s", str.c_str());
506 }
507 else
508 {
509 DrawFormatStringToHandle(
510 text_pos_x,
511 text_pos_y_min + text_interval_y * (text_line++),
512 text_color,
513 font_handle_,
514 " 実際の可動域の内です．");
515 }
516 }
