20void RobotStateNode::ChangeGlobalCenterOfMass(const Vector3& new_com, const bool base)
21{
22 const Vector3 dif = new_com - center_of_mass_global_coord;
23 const Vector3 com_dif = RotateVector3(dif, posture.GetConjugate());
24
25 center_of_mass_global_coord = new_com;
26
27 for (int i = 0; i < HexapodConst::kLegNum; i++)
28 {
29 if (leg_func::IsGrounded(leg_state, i))
30 {
31 leg_pos[i] -= com_dif;
32 leg_reference_pos[i] = leg_pos[i];
33 }
34 else
35 {
36 if (base)leg_reference_pos[i] -= com_dif;
37 }
38 }

41void RobotStateNode::ChangePosture(
42 const std::shared_ptr<const IHexapodCoordinateConverter>& converter_ptr,
43 const Quaternion& new_posture)
44{
45 const Quaternion dif = new_posture * posture.GetConjugate();
46
47 posture = new_posture;
48
49 for (int i = 0; i < HexapodConst::kLegNum; i++)
50 {
51 if (leg_func::IsGrounded(leg_state, i))
52 {
53 Vector3 leg_pos_robot_coord =
54 converter_ptr->ConvertLegToRobotCoordinate(leg_pos[i], i);
55 leg_pos_robot_coord = RotateVector3(leg_pos_robot_coord, dif.GetConjugate());
56 leg_pos[i] = converter_ptr->ConvertRobotToLegCoordinate(leg_pos_robot_coord, i);
57
58 Vector3 leg_reference_pos_robot_coord =
59 converter_ptr->ConvertLegToRobotCoordinate(leg_reference_pos[i], i);
60 leg_reference_pos_robot_coord =
61 RotateVector3(leg_reference_pos_robot_coord, dif.GetConjugate());
62 leg_reference_pos[i] =
63 converter_ptr->ConvertRobotToLegCoordinate(leg_reference_pos_robot_coord, i);
64 }
65 }
66}

145RobotStateNode RobotStateNode::FromString(const std::string& str)
146{
147 std::vector<std::string> data = string_util::Split(str, ",");
148
149 RobotStateNode res;
150 int cnt = 0;
151
152 try
153 {
154 // 脚状態のビット列．
155 res.leg_state = std::bitset<leg_func::kLegStateBitNum>(data[cnt++]);
156
157 // 脚の位置．
158 for (int i = 0; i < HexapodConst::kLegNum; i++)
159 {
160 res.leg_pos[i] =
161 Vector3{ std::stof(data[cnt++]), std::stof(data[cnt++]), std::stof(data[cnt++]) };
162 }
163
164 // 脚の基準位置．
165 for (int i = 0; i < HexapodConst::kLegNum; i++)
166 {
167 res.leg_reference_pos[i] =
168 Vector3{ std::stof(data[cnt++]), std::stof(data[cnt++]), std::stof(data[cnt++]) };
169 }
170
171 // 重心位置．
172 res.center_of_mass_global_coord =
173 Vector3{ std::stof(data[cnt++]), std::stof(data[cnt++]), std::stof(data[cnt++]) };
174
175 // 回転姿勢．
176 res.posture =
177 Quaternion{ std::stof(data[cnt++]),
178 std::stof(data[cnt++]), std::stof(data[cnt++]), std::stof(data[cnt++]) };
179
180 // 次動作．
181 res.next_move = magic_enum::enum_cast<HexapodMove>(data[cnt++]).value();
182
183 // 親ノードの番号．
184 res.parent_index = std::stoi(data[cnt++]);
185
186 // 深さ．
187 res.depth = std::stoi(data[cnt++]);
188 }
189 catch (...)
190 {
192 }
193
194 return res;
195}
