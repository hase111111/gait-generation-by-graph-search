24bool FileTree::SelectFile(const std::string& path, int max_depth,
25 const std::string& extension, const std::string keyword,
26 std::string* output) const
27{
28 assert(output != nullptr);
29
30 // ファイルツリーを作成
31 FileTreeData tree = MakeFileTree(path, max_depth, extension, keyword);
32
33 // ファイルツリーを表示
34 CmdIOUtil::OutputHorizontalLine("*", OutputDetail::kSystem);
35
36 int count = 0;
37 OutputFileTree(tree, 0, true, &count);
38 CmdIOUtil::OutputNewLine(1, OutputDetail::kSystem);
39 CmdIOUtil::OutputHorizontalLine("*", OutputDetail::kSystem);
40 CmdIOUtil::OutputNewLine(1, OutputDetail::kSystem);
41
42 // ファイルを選択
43 std::vector<std::string> file_list = MakeFileList(tree);
44
45 if (file_list.empty())

47 CmdIOUtil::Output("The file did not exist.", OutputDetail::kSystem);
48 return false;
49 }
50
51 while (true)
52 {
53 int select_index =
54 CmdIOUtil::InputInt(0, static_cast<int>(file_list.size()) - 1, 0,
55 "Please select a file. Enter an integer.");
56
57 CmdIOUtil::OutputNewLine(1, OutputDetail::kSystem);
58 CmdIOUtil::Output("The selected file is " + file_list[static_cast<size_t>(select_index)],
59 OutputDetail::kSystem);
60
61 if (CmdIOUtil::InputYesNo("Are you sure?"))
62 {
63 *output = file_list[static_cast<size_t>(select_index)];
64 break;
65 }
66
67 CmdIOUtil::OutputNewLine(1, OutputDetail::kSystem);
68 }
69
70 return true;

73FileTree::FileTreeData FileTree::MakeFileTree(const std::string& path, int max_depth,
74 const std::string& extension,
75 const std::string keyword) const
76{
77 FileTreeData tree;
78
79 tree.path = path;
80
81 assert(std::filesystem::exists(tree.path));
82
83 for (const auto& entry : std::filesystem::directory_iterator(path))
84 {
85 if (entry.is_directory())
86 {
87 // ディレクトリの場合、再帰的にファイルツリーを作成
88 if (max_depth == 0)
89 {
90 tree.directory.push_back(
91 FileTreeData{ entry.path().string(), {}, {} });
92 }
93 else
94 {
95 tree.directory.push_back(
96 MakeFileTree(entry.path().string(), max_depth - 1, extension, keyword));
97 }
98 }
99 else if (entry.is_regular_file())
100 {
101 // ファイルの場合、ファイル名を追加
102
103 if (!extension.empty())
104 {
105 if (!(entry.path().extension().string() == extension ||
106 entry.path().extension().string() == "." + extension))
107 {
108 continue;
109 }
110 }
111
112 if (!keyword.empty())
113 {
114 if (entry.path().filename().string().find(keyword) == std::string::npos)
115 {
116 continue;
117 }
118 }
119
120 tree.file.push_back(entry.path().filename().string());
121 }
122 }
123
124 return tree;
125}

127void FileTree::OutputFileTree(const FileTreeData& tree, int depth,
128 bool not_display_empty, int* file_count) const
129{
130 assert(file_count != nullptr);
131
132 std::string indent = "";
133 for (int i = 0; i < depth; ++i)
134 {
135 indent += "| ";
136 }
137
138 // 空 かつ 空のディレクトリを表示しない設定 でないならば表示する
139 if (!(not_display_empty && tree.file.empty() && tree.directory.empty()))
140 {
141 // ディレクトリ名を出力する際に，パスの階層を削除する
142
143 CmdIOUtil::Output(indent, OutputDetail::kSystem);
144
145 std::string::size_type pos = tree.path.find_last_of("/\\");
146 std::string dir_name = ((depth == 0) ? "" : "- ");
147 dir_name += std::string("[ ") + tree.path.substr(pos + 1) + std::string(" ]");
148
149 CmdIOUtil::Output(indent + dir_name, OutputDetail::kSystem);
150 }
151
152 for (const auto& directory : tree.directory)
153 {
154 OutputFileTree(directory, depth + 1, not_display_empty, file_count);
155 }
156
157 if (!tree.file.empty())
158 {
159 CmdIOUtil::Output(indent + "|", OutputDetail::kSystem);
160
161 for (const auto& file : tree.file)
162 {
163 CmdIOUtil::Output(indent + "|- " + file + " [-" + std::to_string(*file_count) + "-]",
164 OutputDetail::kSystem);
165
166 (*file_count)++;
167 }
168 }
169}

171std::vector<std::string> FileTree::MakeFileList(const FileTreeData& tree) const
172{
173 std::vector<std::string> file_list;
174
175 for (const auto& directory : tree.directory)
176 {
177 std::vector<std::string> tmp = MakeFileList(directory);
178
179 file_list.insert(file_list.end(), tmp.begin(), tmp.end());
180 }
181
182 for (const auto& file : tree.file)
183 {
184 file_list.push_back(tree.path + "\\" + file);
185 }
186
187 return file_list;
188}
