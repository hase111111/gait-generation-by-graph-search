45void DrawCube3D(const VECTOR& center_pos,
46 const float side_len, const unsigned int color)
47{
48 // 立方体は8つの頂点を持つので，それらの座標を計算する．
49 const std::array<VECTOR, 8> vertex =
50 {
51 VGet(center_pos.x - side_len / 2, center_pos.y - side_len / 2,
52 center_pos.z - side_len / 2),
53 VGet(center_pos.x + side_len / 2, center_pos.y - side_len / 2,
54 center_pos.z - side_len / 2),
55 VGet(center_pos.x + side_len / 2, center_pos.y - side_len / 2,
56 center_pos.z + side_len / 2),
57 VGet(center_pos.x - side_len / 2, center_pos.y - side_len / 2,
58 center_pos.z + side_len / 2),
59 VGet(center_pos.x - side_len / 2, center_pos.y + side_len / 2,
60 center_pos.z - side_len / 2),
61 VGet(center_pos.x + side_len / 2, center_pos.y + side_len / 2,
62 center_pos.z - side_len / 2),
63 VGet(center_pos.x + side_len / 2, center_pos.y + side_len / 2,
64 center_pos.z + side_len / 2),
65 VGet(center_pos.x - side_len / 2, center_pos.y + side_len / 2,
66 center_pos.z + side_len / 2)
67 };
68
69 // 3D描画の関数は3角形を基本単位とするので，
70 // 4角形の面を張りたい場合は，2つの三角形を組み合わせる必要がある．
71 // つまり，6面×2つ＝12個の三角形で立方体が描画できる．
72
73 DrawTriangle3D(vertex[0], vertex[1], vertex[2], color, TRUE);
74 DrawTriangle3D(vertex[2], vertex[3], vertex[0], color, TRUE);
75
76 DrawTriangle3D(vertex[4], vertex[5], vertex[6], color, TRUE);
77 DrawTriangle3D(vertex[6], vertex[7], vertex[4], color, TRUE);
78
79 DrawTriangle3D(vertex[4], vertex[7], vertex[0], color, TRUE);
80 DrawTriangle3D(vertex[0], vertex[7], vertex[3], color, TRUE);
81
82 DrawTriangle3D(vertex[1], vertex[2], vertex[5], color, TRUE);
83 DrawTriangle3D(vertex[5], vertex[6], vertex[2], color, TRUE);
84
85 DrawTriangle3D(vertex[0], vertex[1], vertex[5], color, TRUE);
86 DrawTriangle3D(vertex[5], vertex[4], vertex[0], color, TRUE);
87
88 DrawTriangle3D(vertex[2], vertex[3], vertex[7], color, TRUE);
89 DrawTriangle3D(vertex[7], vertex[6], vertex[2], color, TRUE);
90}

111void DrawHexagonalPrism(const std::array<VECTOR, 6>& vertex, const float height,
112 const unsigned int color)
113{
114 // 6角形面の法線方向のベクトルを取得する．
115 // やっている処理としては，頂点0から1へ行くベクトルをv01，
116 // 同様に頂点0から2へ行くベクトルをv02とすると，
117 // v01とv02の外積(Cross)をとると法線方向のベクトルが取得できるため，
118 // これを単位ベクトルに変換(Norm，ノーマライズのこと)し，高さの半分だけ倍にした．
119 const VECTOR center_to_top = VScale(VNorm(VCross(VSub(vertex[0], vertex[1]),
120 VSub(vertex[0], vertex[2]))), height / 2.0f);
121
122 // 上面の頂点．
123 const std::array<VECTOR, 6> vertex_top =
124 {
125 VAdd(vertex[0], center_to_top),
126 VAdd(vertex[1], center_to_top),
127 VAdd(vertex[2], center_to_top),
128 VAdd(vertex[3], center_to_top),
129 VAdd(vertex[4], center_to_top),
130 VAdd(vertex[5], center_to_top)
131 };
132
133 // 底面の頂点．
134 const std::array<VECTOR, 6> vertex_bottom =
135 {
136 VSub(vertex[0], center_to_top),
137 VSub(vertex[1], center_to_top),
138 VSub(vertex[2], center_to_top),
139 VSub(vertex[3], center_to_top),
140 VSub(vertex[4], center_to_top),
141 VSub(vertex[5], center_to_top)
142 };
143
144 DrawHexagon(vertex_top, color); // 上面を描画する．
145 DrawHexagon(vertex_bottom, color); // 底面を描画する．
146
147 // 側面を描画していく．側面は四角形6つで構成されるので，3角形が12個必要になる．
148 for (int i = 0; i < 6; i++)
149 {
150 DrawTriangle3D(vertex_top[i % 6],
151 vertex_top[(i + 1) % 6],
152 vertex_bottom[i % 6],
153 color,
154 TRUE);
155
156 DrawTriangle3D(vertex_top[(i + 1) % 6],
157 vertex_bottom[i % 6],
158 vertex_bottom[(i + 1) % 6],
159 color,
160 TRUE);
161 }
162}
