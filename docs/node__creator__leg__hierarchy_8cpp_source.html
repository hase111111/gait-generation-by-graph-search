23void NodeCreatorLegHierarchy::Create(
24 const RobotStateNode& current_node,
25 const int current_node_index,
26 std::vector<RobotStateNode>* output_graph) const
27{
28 // 現在，接地している脚の本数を数える．
29 const int kLiftedLegNum = leg_func::GetLiftedLegNum(current_node.leg_state);
30
31 // 遊脚している脚の本数によって処理をする．
32 if (kLiftedLegNum == 1)
33 {
34 // 1 本遊脚している．
35 Create1LegLifted(current_node, current_node_index, output_graph);
36 }
37 else if (kLiftedLegNum == 2)
38 {
39 // 2 本遊脚している．
40 Create2LegLifted(current_node, current_node_index, output_graph);
41 }
42 else if (kLiftedLegNum == 3)
43 {
44 // 3 本遊脚している．
45 Create3LegLifted(current_node, current_node_index, output_graph);
46 }
47 else
48 {
49 // ここに来るのは接地している脚の数が6本 or 1本 or 2本の時．
50 // 地面についている脚が3本を切ることはない，何故ならロボットが倒れてしまうため．
51 // また6本接地しているならば脚を動かせない(遊脚する必要がある)．
52 // よって処理を行わない．(そのままの状態を次のノードにする．)
53 RobotStateNode new_node = current_node;
54
55 // 次のノード用に，深さ・親・次の動作を更新する．
56 new_node.ChangeToNextNode(current_node_index, next_move_);
57
58 // 追加する．
59 (*output_graph).emplace_back(new_node);
60 }
61}

66void NodeCreatorLegHierarchy::Create1LegLifted(
67 const RobotStateNode& current_node,
68 const int current_node_index,
69 std::vector<RobotStateNode>* output_graph) const
70{
71 // 遊脚している脚を探す．遊脚数は1なので1つの数字が帰るはず
72 std::vector<int> lifted_leg_list;
73
74 leg_func::GetLiftedLegIndexByVector(current_node.leg_state, &lifted_leg_list);
75
76
77 // 列挙体 DiscreteLegPos の全ての要素でループを回す．
78 for (const auto i : discrete_leg_pos_list_)
79 {
80 RobotStateNode new_node = current_node; // 新しい脚状態を生成する.
81
82 // 脚状態を変更する．
83 leg_func::ChangeDiscreteLegPos(lifted_leg_list[0], i, &new_node.leg_state);
84
85 // 次のノード用に，深さ・親・次の動作を更新する．
86 new_node.ChangeToNextNode(current_node_index, next_move_);
87
88 (*output_graph).emplace_back(new_node); // 追加する．
89 }
90}
