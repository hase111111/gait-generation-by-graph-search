58 //胴体を描画する．

60
61 for (int i = 0; i < HexapodConst::kLegNum; i++)

78
79 //脚を描画する．
80 for (int i = 0; i < HexapodConst::kLegNum; i++)
81 {
82 //脚の色を遊脚・接地で変更する．

98 true

100 );
101

105 i,
106 draw_node_.center_of_mass_global_coord,
107 draw_node_.posture,
108 true
109 )
110 );
111
112 if (draw_joint_state_[i].is_in_range)

121
122
123 //間接の描画
124 for (int j = 0; j < 4; j++)
125 {
126 unsigned int color = kJointColor;
127
128 if (j == 0 && !PhantomXMkIIConst::IsValidCoxaAngle(i, draw_joint_state_[i].joint_angle[0])) { color = kColorErrorJoint; }
129 if (j == 1 && !PhantomXMkIIConst::IsValidFemurAngle(draw_joint_state_[i].joint_angle[1])) { color = kColorErrorJoint; }
130 if (j == 2 && !PhantomXMkIIConst::IsValidTibiaAngle(draw_joint_state_[i].joint_angle[2])) { color = kColorErrorJoint; }
131
132 VECTOR pos = dxlib_util::ConvertToDxlibVec(
133 converter_ptr_->ConvertLegToGlobalCoordinate(
134 draw_joint_state_[i].joint_pos_leg_coordinate[j],
135 i,
136 draw_node_.center_of_mass_global_coord,
137 draw_node_.posture,
138 true
139 )
140 );
141
142 DrawSphere3D(pos, kJointRadius, kSphereDivNum, color, color, TRUE);
143 }
144
145 DrawSphere3D(
146 dxlib_util::ConvertToDxlibVec(converter_ptr_->ConvertLegToGlobalCoordinate(draw_node_.leg_pos[i], i, draw_node_.center_of_mass_global_coord, draw_node_.posture, true)),
147 kJointRadius / 2,
148 kSphereDivNum,
149 kJointColor,
150 kJointColor,
151 TRUE
152 );
153
154 //脚の接地の基準地点の描画
155 DrawSphere3D(

157 kJointRadius / 3,
158 kSphereDivNum,
159 kColorLegBase,
160 kColorLegBase,
161 TRUE
162 );
163 }
164}
