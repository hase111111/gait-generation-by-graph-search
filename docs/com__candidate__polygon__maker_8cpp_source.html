21void ComCandidatePolygonMaker::MakeCandidatePolygon(
22 const RobotStateNode& node,
23 std::array<ComPosAndPolygon, kMakePolygonNum>* output_poly) const
24{
25 assert(converter_ptr_ != nullptr); // nullptr でないことを確認する．
26 assert(output_poly != nullptr); // nullptr でないことを確認する．
27
28 // XY平面に射影した脚位置を算出する．(グローバル座標系)
29 std::array<Vector2, HexapodConst::kLegNum> leg_pos_xy;
30
31 for (int i = 0; i < HexapodConst::kLegNum; i++)
32 {
33 // 脚位置(グローバル座標)をXY平面に射影する．
34 leg_pos_xy[i] = converter_ptr_->ConvertLegToGlobalCoordinate(
35 node.leg_pos[i],
36 i,
37 node.center_of_mass_global_coord,
38 node.posture,
39 true).ProjectedXY();
40 }
41
42 // 中心を囲むように4角形を作成する．
43 MakeCandidateBox(leg_pos_xy, 0, &(*output_poly)[0].polygon);
44 (*output_poly)[0].com_pos = enums::DiscreteComPos::kFrontLeft;
45 (*output_poly)[0].is_able = true;
46
47 MakeCandidateBox(leg_pos_xy, 1, &(*output_poly)[1].polygon);
48 (*output_poly)[1].com_pos = enums::DiscreteComPos::kBackLeft;
49 (*output_poly)[1].is_able = true;
50
51 MakeCandidateBox(leg_pos_xy, 2, &(*output_poly)[2].polygon);
52 (*output_poly)[2].com_pos = enums::DiscreteComPos::kBack;
53 (*output_poly)[2].is_able = true;
54
55 MakeCandidateBox(leg_pos_xy, 3, &(*output_poly)[3].polygon);
56 (*output_poly)[3].com_pos = enums::DiscreteComPos::kBackRight;
57 (*output_poly)[3].is_able = true;
58
59 MakeCandidateBox(leg_pos_xy, 4, &(*output_poly)[4].polygon);
60 (*output_poly)[4].com_pos = enums::DiscreteComPos::kFrontRight;
61 (*output_poly)[4].is_able = true;
62
63 MakeCandidateBox(leg_pos_xy, 5, &(*output_poly)[5].polygon);
64 (*output_poly)[5].com_pos = enums::DiscreteComPos::kFront;
65 (*output_poly)[5].is_able = true;
66
67 // 中心に3角形を作成する．
68 MakeCandidateTriangle(leg_pos_xy, &(*output_poly)[6]);
69
70 // 生成した多角形が正しいかどうかをチェックし，異常なものは削除する．
71 if (kDoCheckPolygon)
72 {
73 for (int i = 0; i < kMakePolygonNum; ++i)
74 {
75 if (!IsAblePolygon((*output_poly)[i].polygon))
76 {
77 (*output_poly)[i].is_able = false;
78 }
79 }
80 }
81}

83void ComCandidatePolygonMaker::MakeCandidateBox(
84 const std::array<Vector2, HexapodConst::kLegNum>& leg_pos,
85 const int start_leg_num,
86 Polygon2* output_poly) const
87{
88 // 脚位置を線で結ぶ．この交点から重心候補地点が存在する多角形を求める．
89 LineSegment2 leg_line_02(
90 leg_pos[(start_leg_num + 0) % HexapodConst::kLegNum],
91 leg_pos[(start_leg_num + 2) % HexapodConst::kLegNum]);
92
93 LineSegment2 leg_line_03(
94 leg_pos[(start_leg_num + 0) % HexapodConst::kLegNum],
95 leg_pos[(start_leg_num + 3) % HexapodConst::kLegNum]);
96
97 LineSegment2 leg_line_14(
98 leg_pos[(start_leg_num + 1) % HexapodConst::kLegNum],
99 leg_pos[(start_leg_num + 4) % HexapodConst::kLegNum]);
100
101 LineSegment2 leg_line_15(
102 leg_pos[(start_leg_num + 1) % HexapodConst::kLegNum],
103 leg_pos[(start_leg_num + 5) % HexapodConst::kLegNum]);
104
105 LineSegment2 leg_line_25(
106 leg_pos[(start_leg_num + 2) % HexapodConst::kLegNum],
107 leg_pos[(start_leg_num + 5) % HexapodConst::kLegNum]);
108
109 // 交点(intersection)を求める．
110 Vector2 intersection_02_14 = leg_line_02.GetIntersection(leg_line_14);
111 Vector2 intersection_02_15 = leg_line_02.GetIntersection(leg_line_15);
112 Vector2 intersection_03_14 = leg_line_03.GetIntersection(leg_line_14);
113 Vector2 intersection_03_15 = leg_line_03.GetIntersection(leg_line_15);
114
115 // 中心と0番の脚位置を結んだ線分を求める．
116 LineSegment2 leg_line_0_center(
117 leg_pos[(start_leg_num + 0) % HexapodConst::kLegNum], intersection_03_14);
118
119 // 多角形生成．
120
121 (*output_poly).Reset();
122
123 if (leg_line_0_center.HasIntersection(leg_line_25))
124 {
125 // 交点がある場合，5角形の多角形を作成する．
126 Vector2 intersection_03_25 = leg_line_03.GetIntersection(leg_line_25);
127 Vector2 intersection_14_25 = leg_line_14.GetIntersection(leg_line_25);
128
129 (*output_poly).AddVertexCheckForDuplicates(intersection_03_15);
130 (*output_poly).AddVertexCheckForDuplicates(intersection_02_15);
131 (*output_poly).AddVertexCheckForDuplicates(intersection_02_14);
132 (*output_poly).AddVertexCheckForDuplicates(intersection_14_25);
133 (*output_poly).AddVertexCheckForDuplicates(intersection_03_25);
134 }
135 else
136 {
137 // 交点がない場合，既に求めた4点で，4角形の多角形を作成する．
138 (*output_poly).AddVertex(intersection_03_15);
139 (*output_poly).AddVertex(intersection_02_15);
140 (*output_poly).AddVertex(intersection_02_14);
141 (*output_poly).AddVertex(intersection_03_14);
142 }
143}

145void ComCandidatePolygonMaker::MakeCandidateTriangle(
146 const std::array<Vector2, HexapodConst::kLegNum>& leg_pos,
147 ComPosAndPolygon* output) const
148{
149 LineSegment2 leg_line_03(leg_pos[0], leg_pos[3]);
150 LineSegment2 leg_line_14(leg_pos[1], leg_pos[4]);
151 LineSegment2 leg_line_25(leg_pos[2], leg_pos[5]);
152
153 // 交点(intersection)を求める．
154 Vector2 intersection_03_14 = leg_line_03.GetIntersection(leg_line_14);
155 Vector2 intersection_03_25 = leg_line_03.GetIntersection(leg_line_25);
156 Vector2 intersection_14_25 = leg_line_14.GetIntersection(leg_line_25);
157
158 // 三角形を作成する．
159 output->polygon.Reset();
160 output->polygon.AddVertex(intersection_03_14);
161 output->polygon.AddVertex(intersection_03_25);
162 output->polygon.AddVertex(intersection_14_25);
163
164 if (intersection_03_14.x > intersection_03_25.x)
165 {
166 output->com_pos = enums::DiscreteComPos::kCenterBack;
167 }
168 else
169 {
170 output->com_pos = enums::DiscreteComPos::kCenterFront;
171 }
172
173 output->is_able = true;
174
175 return;
176}

178bool ComCandidatePolygonMaker::IsAblePolygon(const Polygon2& poly) const
179{
180 // 生成されるのは 3 or 4 or 5角形のみ．
181 if (poly.GetVertexNum() == 3 ||
182 poly.GetVertexNum() == 4 ||
183 poly.GetVertexNum() == 5)
184 {
185 // 凸多角形であるかを確認する．
186 if (poly.IsConvex())
187 {
188 return true;
189 }
190 }
191
192 return false;
193}
