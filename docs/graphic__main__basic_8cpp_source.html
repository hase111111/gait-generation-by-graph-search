82bool GraphicMainBasic::Update()
83{
84 mouse_ptr_->Update();
85
86 if (map_update_count != broker_ptr_->map_state.GetUpdateCount())
87 {
88 map_update_count = broker_ptr_->map_state.GetUpdateCount();
89 map_state_ = broker_ptr_->map_state.GetData();
90
91 map_renderer_ptr_->SetMapState(map_state_);
92 }
93
94
95 // ノードを読み出す時間になったら，仲介人からデータを読み出す．
96 if (counter_ % kNodeGetCount == 0 && graph_update_count != broker_ptr_->graph.GetUpdateCount())
97 {
98 // 仲介人からデータを読み出す．
99 graph_ = broker_ptr_->graph.GetData();
100
101 std::vector<size_t> simu_end_index;
102
103 simu_end_index = broker_ptr_->simulation_end_index.GetData();
104
105 // ノードの情報を表示するGUIに情報を伝達する．
106 display_node_switch_gui_->SetGraphData(graph_.size(), simu_end_index);
107
108
109 // 移動軌跡を更新する．
110 movement_locus_renderer_.SetMoveLocusPoint(graph_);
111
112 movement_locus_renderer_.SetSimulationEndIndexes(simu_end_index);
113
114
115 // ロボットの接地点を更新する．
116 robot_ground_point_renderer_.SetNodeAndSimulationEndNodeIndex(graph_, simu_end_index);
117
118
119 graph_update_count = broker_ptr_->graph.GetUpdateCount();
120 }
121
122
123 // ノードが存在しているのならば，各クラスに情報を伝達する
124 if (!graph_.empty())
125 {
126 // 表示ノードが更新されたら，表示するノードを変更する．
127 if (display_node_index_ != display_node_switch_gui_->GetDisplayNodeNum())
128 {
129 if (display_node_index_ > 0)
130 {
131 // 0番目のアニメーションを指定する．
132 interpolated_anime_start_count_ = 0;
133
134 interpolated_node_.clear();
135
136 interpolated_node_ = interpolated_node_creator_.CreateInterpolatedNode(
137 graph_[display_node_index_],
138 graph_[display_node_switch_gui_->GetDisplayNodeNum()]);
139 }
140
141 // 表示するノードを取得する．
142 display_node_index_ = display_node_switch_gui_->GetDisplayNodeNum();
143 display_node_switch_gui_->InAnimation(true);
144
145 // ノードの情報を設定するGUIに情報を伝達する．
146 node_setter_group_.SetNode(graph_.at(display_node_index_));
147 }
148
149 if (0 < interpolated_node_.size() &&
150 0 <= interpolated_anime_start_count_ &&
151 interpolated_anime_start_count_ < interpolated_node_.size())
152 {
153 // アニメーション中は interpolated_node_ の補完されたノードを表示する．
154 if (counter_ % kInterpolatedAnimeCount == 0)
155 {
156 interpolated_anime_start_count_ += display_node_switch_gui_->GetAnimeSpeed();
157 }
158
159 if (interpolated_anime_start_count_ < interpolated_node_.size())
160 {
161 node_setter_group_.SetNode(interpolated_node_[interpolated_anime_start_count_]);
162 }
163 }
164 else if (interpolated_node_.size() <= interpolated_anime_start_count_)
165 {
166 display_node_switch_gui_->InAnimation(false);
167 interpolated_anime_start_count_ = -1;
168 node_setter_group_.SetNode(graph_.at(display_node_index_));
169 }
170 }
171
172 ++counter_; // カウンタを進める．
173
174 gui_updater_.Activate(mouse_ptr_); // GUIをアクティブにする．
175
176 return true;
177}
