97 constexpr void SetMapSize(const float max_x, const float min_x, const float max_y,
98 const float min_y, const float start_rough_x, const float map_base_z)
99 {
100 assert(min_x < max_x);
101 assert(min_y < max_y);
102 assert(min_x < start_rough_x);
103 assert(start_rough_x < max_x);
104
105 map_max_x = max_x;
106 map_min_x = min_x;
107 map_max_y = max_y;
108 map_min_y = min_y;
109 map_start_rough_x = start_rough_x;
110 base_z = map_base_z;
111 }

115 assert(0 < interval);

117 stripe_interval = interval;
118 }

122 assert(0 <= rate);

125 hole_rate = rate;
126 }
127

137 {
138 tilt_angle = angle;

140
141 constexpr void SetSlopeAngle(const float angle)
142 {
143 slope_angle = angle;
144 }
145
146 constexpr void SetRouthHeight(const float height_min, const float height_max)

149
150 rough_min_height = height_min;
151 rough_max_height = height_max;
152 }
