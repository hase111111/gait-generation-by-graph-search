25class TestMapCreator final : public IMapCreator
26{
27public:
28 static constexpr float kMax{ 1000.f };
29 static constexpr float kMin{ -kMax };
30 static constexpr float kInterval{ 10.f };
31
33 MapState InitMap() override
34 {
35 std::vector<Vector3> map_point;
36
37 for (float x = kMin; x <= kMax; x += kInterval)
38 {
39 for (float y = kMin; y <= kMax; y += kInterval)
40 {
41 Vector3 point(x, y, 0);
42 map_point.emplace_back(point);
43 }
44 }
45
46 return MapState(map_point);
47 }

51 }

63TEST_SUITE("DividedMapState::Init")
64{
65 using designlab::DividedMapState;
66 using designlab::MapState;
67 using designlab::Vector3;
68
69 TEST_CASE("マップの状態とロボットの重心座標を渡した時，マップを重心を中心に格子状に分割するべき")
70 {
71 using divided_map_state_test_internal::TestMapCreator;
72
73 // Act.テスト用のマップを生成する．
74 TestMapCreator test_map_creator;
75 MapState map_state = test_map_creator.InitMap();
76
77 // Act.テスト用のロボットの重心座標を生成する．
78 const Vector3 global_robot_com(0, 0, 0);
79
80 // Arrange.テスト対象のクラスのメソッドを呼び出す．
81 DividedMapState divided_map_state;
82 divided_map_state.Init(map_state, global_robot_com);
83
84 // Assert.マップのサイズが正しいか確認する．
85 int center_index_x = divided_map_state.GetDividedMapIndexX(global_robot_com.x);
86 int center_index_y = divided_map_state.GetDividedMapIndexX(global_robot_com.y);
87
88 constexpr int expected_num = static_cast<int>(
89 (DividedMapState::kDividedAreaLength / TestMapCreator::kInterval) *
90 (DividedMapState::kDividedAreaLength / TestMapCreator::kInterval));
91
92 CHECK_EQ(divided_map_state.GetPointNum(center_index_x, center_index_y), expected_num);
93 }
94}
