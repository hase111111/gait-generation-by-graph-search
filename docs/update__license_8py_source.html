35def update_license(filepath, new_license):

41 # `//` で始まらない最初の行を探す
42 start_index = 0
43 for i, line in enumerate(lines):
44 if not line.strip().startswith("//"): # `//` で始まらない行を見つける
45

54 new_content = new_license + "\n" + "".join(lines[start_index:])
