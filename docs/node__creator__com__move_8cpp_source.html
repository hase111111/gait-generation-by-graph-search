32void NodeCreatorComMove::Create(const RobotStateNode& current_node,
33 const int current_num,
34 std::vector<RobotStateNode>* output_graph) const
35{
36 std::array<ComPosAndPolygon, ComCandidatePolygonMaker::kMakePolygonNum> candidate_polygons;
37
38 // 重心移動先の候補地点の範囲を示す多角形を作成する．
39 maker_.MakeCandidatePolygon(current_node, &candidate_polygons);
40
41 // 候補範囲から実際に移動する先の座標を選択する．
42 for (int i = 0; i < ComCandidatePolygonMaker::kMakePolygonNum; ++i)
43 {
44 // そもそも多角形が候補点になりえないならば，その多角形は無視する．
45 if (!candidate_polygons[i].is_able) { continue; }
46
47 Vector3 result_com;
48
49 if (selector_.GetComFromPolygon(candidate_polygons[i].polygon, current_node, &result_com))
50 {
51 RobotStateNode next_node = current_node;
52
53 // 重心位置を変更し，それに伴い接地脚の位置も変更する．
54 next_node.ChangeGlobalCenterOfMass(result_com, false);
55
56 // leg_state の com_pattern を変更する
57 leg_func::ChangeDiscreteComPos(candidate_polygons[i].com_pos, &next_node.leg_state);
58
59 for (int j = 0; j < HexapodConst::kLegNum; ++j)
60 {
61 leg_func::ChangeDiscreteLegPos(j, enums::DiscreteLegPos::kCenter,
62 &next_node.leg_state);
63 }
64
65 next_node.ChangeToNextNode(current_num, next_move_); // 深さや親ノードを変更する
66
67 if (checker_ptr_->IsStable(next_node.leg_state, next_node.leg_pos) &&
68 !checker_ptr_->IsBodyInterferingWithGround(next_node, map_))
69 {
70 (*output_graph).push_back(next_node);
71 }
72 }
73 }
74
75 // 結果が空だったら現在姿勢をそのまま追加する．
76 if ((*output_graph).empty())
77 {
78 RobotStateNode next_node = current_node;
79 next_node.ChangeToNextNode(current_num, next_move_);
80
81 (*output_graph).push_back(next_node);
82 }
83}
