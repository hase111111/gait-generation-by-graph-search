35void NodeCreatorComUpDown::Create(const RobotStateNode& current_node, const int current_num,
36 std::vector<RobotStateNode>* output_graph) const
37{
38 // 重心を最も高くあげることのできる位置と，最も低く下げることのできる位置を求める．
39
40 // マップの最大z座標を求める．
42 float map_highest_z = -100000;
43
44 if (map_.IsInMap(current_node.center_of_mass_global_coord))
45 {
46 const int map_index_x = map_.GetDividedMapIndexX(current_node.center_of_mass_global_coord.x);
47 const int map_index_y = map_.GetDividedMapIndexY(current_node.center_of_mass_global_coord.y);
48 map_highest_z = map_.GetTopZ(map_index_x, map_index_y);
49 }
50
51 for (int i = 0; i < HexapodConst::kLegNum; i++)
52 {
53 // 脚の先端の座標を求める．
54 const Vector3 kCoxaVec = converter_ptr_->ConvertRobotToGlobalCoordinate(
55 presenter_ptr_->GetLegBasePosRobotCoordinate(i),
56 current_node.center_of_mass_global_coord,
57 current_node.posture, true);
58
59 if (map_.IsInMap(kCoxaVec))
60 {
61 const int kCoxaX = map_.GetDividedMapIndexX(kCoxaVec.x);
62 const int kCoxaY = map_.GetDividedMapIndexY(kCoxaVec.y);
63 map_highest_z = (std::max)(map_.GetTopZ(kCoxaX, kCoxaY), map_highest_z);
64 }
65 }
66
67
68 // ロボットの重心の最も低く下げることのできるz座標と，
69 // 高くあげることができるz座標を求める．どちらもグローバル座標．
70 float highest_body_pos_z = map_highest_z + presenter_ptr_->GetGroundHeightMarginMax();
71 float lowest_body_pos_z = map_highest_z + presenter_ptr_->GetGroundHeightMarginMin();
72
73
74 // 最も高い地点を修正する．
75
76 for (int i = 0; i < HexapodConst::kLegNum; i++)
77 {
78 // 接地している脚についてのみ考える．
79 if (leg_func::IsGrounded(current_node.leg_state, i))
80 {
81 // 三平方の定理を使って，脚接地地点から重心位置をどれだけ上げられるか考える．
82 const float edge_c =
83 PhantomXMkIIConst::kFemurLength + PhantomXMkIIConst::kTibiaLength;
84 const float edge_b =
85 current_node.leg_pos[i].ProjectedXY().GetLength() - PhantomXMkIIConst::kCoxaLength;
86
87 const float edge_a = sqrt(math_util::Squared(edge_c) - math_util::Squared(edge_b));
88
89 // 接地脚の最大重心高さの中から一番小さいものを全体の最大重心位置として記録する．_aは脚の接地点からどれだけ上げられるかを表しているので，グローバル座標に変更する．
90 highest_body_pos_z = (std::min)(
91 edge_a + current_node.center_of_mass_global_coord.z + current_node.leg_pos[i].z, highest_body_pos_z);
92 }
93 }
94
95 // ノードを追加する．
96 pushNodeByMaxAndMinPosZ(current_node, current_num, highest_body_pos_z,
97 lowest_body_pos_z, output_graph);
98}

101void NodeCreatorComUpDown::pushNodeByMaxAndMinPosZ(
102 const RobotStateNode& current_node, const int current_num, const float high,
103 const float low, std::vector<RobotStateNode>* output_graph) const
104{
105 // 重心を変化させたものを追加する．変化量が一番少ないノードは削除する．
106
107 // 最大と最小の間を分割する．
108 const float kDivZ = (high - low) / static_cast<float>(kDiscretization);
109
110 // 分割した分新しいノードを追加する．
111 for (int i = 0; i < kDiscretization + 1; ++i)
112 {
113 bool is_valid = true;
114
115 RobotStateNode new_node = current_node;
116
117 // 重心の位置を変更する．
118 Vector3 new_com = current_node.center_of_mass_global_coord;
119 new_com.z = low + kDivZ * i;
120
121 new_node.ChangeGlobalCenterOfMass(new_com, true);
122
123
124 for (int j = 0; j < HexapodConst::kLegNum; j++)
125 {
126 if (!checker_ptr_->IsLegInRange(j, new_node.leg_pos[j]))
127 {
128 is_valid = false;
129 }
130 }
131
132 // current_numを親とする，新しいノードに変更する
133 new_node.ChangeToNextNode(current_num, next_move_);
134
135 // ノードを追加する．
136 if (is_valid)
137 {
138 (*output_graph).emplace_back(new_node);
139 }
140 }
141
142 // 重心の変化が一切ないものを追加する．
143 RobotStateNode same_node = current_node;
144
145 same_node.ChangeToNextNode(current_num, next_move_);
146
147 (*output_graph).emplace_back(same_node);
148}
