12LegStateBit MakeLegStateBit(
13 const enums::DiscreteComPos discrete_com_pos,
14 const std::array<bool, HexapodConst::kLegNum>& is_ground,
15 const std::array<enums::DiscreteLegPos, HexapodConst::kLegNum>& discretized_leg_pos)
16{
17 LegStateBit res = 0;
18
19 // bitに変換する．
20 LegStateBit discrete_com_pos_bit = static_cast<unsigned int>(discrete_com_pos);
21 res |= discrete_com_pos_bit << kShiftToComNum; // 重心パターンの数値だけビットを立てる．
22
23 for (int i = 0; i < HexapodConst::kLegNum; i++)
24 {
25 // 接地しているならば上位ビットを立てる．
26 size_t ground_bit_index = static_cast<size_t>(i + 1) * 4 - 1;
27 if (is_ground[i]) { res[ground_bit_index] = true; }
28
29 // 脚のビットを立てる．
30 LegStateBit discrete_leg_pos_bit = static_cast<unsigned int>(discretized_leg_pos[i]);
31 size_t shift_num = static_cast<size_t>(i) * 4; // 4bitずつずらす
32
33 res |= discrete_leg_pos_bit << shift_num; // 脚の位置の数値だけビットを立てる
34 }
35
36 return res;
37}

40bool IsGrounded(const LegStateBit& leg_state, const int leg_index)
41{
42 // leg_indexは0～5の範囲にある必要がある．
43 assert(0 <= leg_index);
44 assert(leg_index < HexapodConst::kLegNum);
45
46 // 指定された脚の接地脚のビットが立っているか調べる．
47 size_t ground_bit_index = static_cast<size_t>(leg_index + 1) * 4 - 1;
48
49 if (leg_state[ground_bit_index])
50 {
51 return true;
52 }
53 else
54 {
55 return false;
56 }
57}

134enums::DiscreteLegPos GetDiscreteLegPos(const LegStateBit& leg_state, const int leg_index)
135{
136 // leg_indexは0～5の範囲にある必要がある．
137 assert(0 <= leg_index);
138 assert(leg_index < HexapodConst::kLegNum);
139
140 const int shift_num = 4 * leg_index; // 4bitずつずらす
141
142 const int res = static_cast<int>(
143 ((leg_state & (kLegPosMaskBit << shift_num)) >> shift_num).to_ulong());
144

175void ChangeDiscreteLegPos(
176 const int leg_index,
177 const enums::DiscreteLegPos new_discretized_leg_pos,
178 LegStateBit* leg_state
179)
180{
181 // leg_indexは0～5の範囲にある必要がある．
182 assert(0 <= leg_index);
183 assert(leg_index < HexapodConst::kLegNum);
184
185 // leg_state は nullptrではない．
186 assert(leg_state != nullptr);
187
188
189 // 新しい脚状態を生成する．
190 const size_t shift_num = static_cast<size_t>(leg_index) * 4; // 4bitずつずらす．
191
192 // 4bitのデータを変更する地点までマスクをずらす．
193 const LegStateBit mask = kLegPosMaskBit << shift_num;
194
195 // bitに変換する．
196 const LegStateBit discreate_leg_pos_bit = static_cast<unsigned int>(new_discretized_leg_pos);
197
198 // 脚位置のデータは4bitづつ配置されているのでその位置まで移動する．
199 const LegStateBit state = discreate_leg_pos_bit << shift_num;
200
201 // 浮いている脚の脚位置のみを変更（排他的論理和による特定ビットの交換．
202 // 参考 : https://qiita.com/vivisuke/items/bc707190e008551ca07f (アクセス日 2023/12/27)
203 LegStateBit res = ((*leg_state) ^ state) & mask;
204 (*leg_state) ^= res;
205}

218 const size_t ground_bit_index = static_cast<size_t>(leg_index + 1) * 4 - 1;

241void ChangeDiscreteComPos(const enums::DiscreteComPos new_com_pattern, LegStateBit* leg_state)
242{
243 // leg_state は nullptrではない．
