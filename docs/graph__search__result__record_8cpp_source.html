20std::string GraphSearchResultRecord::ToCsvString() const
21{
22 // std::boolalphaを使うと，bool値を文字列に変換できる．
23 // excelにおいて，数値を文字列として扱うためには，
24 // 数値の前にシングルクォーテーション ' をつける必要がある．
25
26 std::stringstream stream;
27
28 stream << "'" << result_node.leg_state.to_string() << ",";
29
30 for (int i = 0; i < HexapodConst::kLegNum; i++)
31 {
32 stream << std::boolalpha << leg_func::IsGrounded(result_node.leg_state, i) << ",";
33 }
34
35 for (int i = 0; i < HexapodConst::kLegNum; i++)
36 {
37 stream << string_util::EnumToStringRemoveTopK(
38 leg_func::GetDiscreteLegPos(result_node.leg_state, i)) << ",";
39 }
40
41 stream << string_util::EnumToStringRemoveTopK(
42 leg_func::GetDiscreteComPos(result_node.leg_state)) << ",";
43
44 stream << result_node.center_of_mass_global_coord << ",";
45 stream << result_node.posture << ",";
46
47 for (int i = 0; i < HexapodConst::kLegNum; i++)
48 {
49 stream << result_node.leg_pos[i] << ",";
50 }
51
52 for (int i = 0; i < HexapodConst::kLegNum; i++)
53 {
54 stream << result_node.leg_reference_pos[i] << ",";
55 }
56
57 stream << string_util::EnumToStringRemoveTopK(result_node.next_move) << ",";
58 stream << math_util::FloatingPointNumToString(computation_time) << ",";
59 stream << graph_search_result.ToString();
60
61 return stream.str();
62}
