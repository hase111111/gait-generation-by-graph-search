44GraphSearchResult GaitPatternGeneratorRevaluation::GetNextNodeByGraphSearch(
45 const RobotStateNode& current_node,
46 const MapState& map_state,
47 const RobotOperation& operation,
48 RobotStateNode* output_node)
49{
50 assert(output_node != nullptr); // output_nodeは nullptrでない
51
52 const GraphSearchResult result =
53 gpg_ptr_->GetNextNodeByGraphSearch(
54 current_node, map_state, operation, output_node);
55
56 if (result.result != enums::Result::kSuccess)
57 {
58 // グラフ探索に失敗した場合は終了．
59 return result;
60 }
61
62 // 成功した場合は，逆運動学計算で脚軌道生成が可能であるか確認する．
63 if (IsValidNode(current_node, *output_node))
64 {
65 // 有効なノードである場合は，そのまま終了．
66 return result;
67 }
68
69 // 逆運動学計算で脚軌道生成が不可能な場合は，再評価を行う．
70 return gpg_revaluation_ptr_->GetNextNodeByGraphSearch(
71 current_node, map_state, operation, output_node);
72}

74bool GaitPatternGeneratorRevaluation::IsValidNode(
75 const RobotStateNode& current_node,
76 const RobotStateNode& next_node) const
77{
78 // まず，next_nodeが有効なノードであるかを確認する．
79 const auto joint = joint_calculator_ptr_->CalculateAllJointState(next_node);
80
81 if (!joint_calculator_ptr_->IsValidAllJointState(next_node, joint))
82 {
83 // 逆運動学計算に失敗した場合は無効なノードとする．
84 return false;
85 }
86
87 // 次に補間ノードを生成する．
88 const std::vector<RobotStateNode> interpolated_node =
89 interpolated_node_creator_.CreateInterpolatedNode(current_node, next_node);
90
91 for (const auto& node : interpolated_node)
92 {
93 const auto joint_interpolated =
94 joint_calculator_ptr_->CalculateAllJointState(node);
95
96 if (!joint_calculator_ptr_->IsValidAllJointState(node, joint_interpolated))
97 {
98 // 逆運動学計算に失敗した場合は無効なノードとする．
99 return false;
100 }
101 }
102
103 return true;
104}
