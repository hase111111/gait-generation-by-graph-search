21 inline char*
22 align_up(char* ptr, std::size_t alignment) noexcept
23 {
24 return reinterpret_cast<char*>(~(alignment - 1) &
25 (reinterpret_cast<uintptr_t>(ptr) + alignment - 1));

105 template <class CharT,class Extra,class Allocator>
106 class heap_string_factory
107 {
108 public:
109 using char_type = CharT;
110 using heap_string_type = heap_string<CharT,Extra,Allocator>;
111 private:
112
113 using byte_allocator_type = typename std::allocator_traits<Allocator>::template rebind_alloc<char>;
114 using byte_pointer = typename std::allocator_traits<byte_allocator_type>::pointer;
115
116 using heap_string_allocator_type = typename std::allocator_traits<Allocator>::template rebind_alloc<heap_string_type>;
117 public:
118 using pointer = typename std::allocator_traits<heap_string_allocator_type>::pointer;
119
120 struct storage_t
121 {
122 heap_string_type data;
123 char_type c[1];
124 };
125 typedef typename jsoncons_aligned_storage<sizeof(storage_t), alignof(storage_t)>::type storage_type;
126
127 static size_t aligned_size(std::size_t n)
128 {
129 return sizeof(storage_type) + n;
130 }
131
132 public:
133
134 static pointer create(const char_type* s, std::size_t length, Extra extra, const Allocator& alloc)
135 {
136 std::size_t len = aligned_size(length*sizeof(char_type));
137
138 std::size_t align = alignof(storage_type);
139 std::size_t mem_len = (align-1)+len;
140
141 byte_allocator_type byte_alloc(alloc);
142 byte_pointer ptr = byte_alloc.allocate(mem_len);
143
144 char* q = extension_traits::to_plain_pointer(ptr);
145
146 char* storage = align_up(q, align);
147
148 JSONCONS_ASSERT(storage >= q);
149
150 heap_string_type* ps = new(storage)heap_string_type(extra, byte_alloc);
151
152 auto psa = launder_cast<storage_t*>(storage);
153
154 CharT* p = new(&psa->c)char_type[length + 1];
155 std::memcpy(p, s, length*sizeof(char_type));
156 p[length] = 0;
157 ps->p_ = std::pointer_traits<typename heap_string_type::pointer>::pointer_to(*p);
158 ps->length_ = length;
159 ps->offset_ = (uint8_t)(storage - q);
160 return std::pointer_traits<pointer>::pointer_to(*ps);
161 }
162
163 static void destroy(pointer ptr)
164 {
165 if (ptr != nullptr)
166 {
167 heap_string_type* rawp = extension_traits::to_plain_pointer(ptr);
168
169 char* q = launder_cast<char*>(rawp);
170
171 char* p = q - ptr->offset_;
172
173 std::size_t mem_size = (alignof(storage_type)-1)+ aligned_size(ptr->length_*sizeof(char_type));
174 byte_allocator_type byte_alloc(ptr->get_allocator());
175 byte_alloc.deallocate(p,mem_size + ptr->offset_);
176 }
177 }
178 };
