49 Matrix mul(const Matrix& A, const Matrix& B) {

58 // O(n^3), Givens Rotation
59#define MAKEROT(x,y,c,s) {double r = sqrt(x*x+y*y); c = x/r; s = y/r;}
60#define ROT(x,y,c,s) {double u = c*x+s*y; double v = -s*x+c*y; x = u; y = v;}
61 Array Givens_rotation(Matrix A, Array b) {
62 int n = (int)b.size();
63 for (int i = 0; i < n; ++i) {
64 for (int j = i + 1; j < n; j++) {
65 double c, s;
66 MAKEROT(A[i][i], A[j][i], c, s);
67 ROT(b[i], b[j], c, s);
68 for (int k = i; k < n; k++)
69 ROT(A[i][k], A[j][k], c, s);
70 }
71 }
72 for (int i = n - 1; i >= 0; i--) {
73 for (int j = i + 1; j < n; j++)
74 b[i] = b[i] - A[i][j] * b[j];
75 b[i] = b[i] / A[i][i];
76 }
77 return b;
78 }

113 double SNE()
114 {
115 double ret = 100000000000;
116
117 if (this->Legs.size() < 3)
118 {
119 std::cerr << "ERROR S_NE !!!!!" << std::endl;
120 return 0.0;
121 }
122
123 //まずは残渣 z = ax + by + cとして基準平面を求める．
124 double a, b, c;
125
126 {
127 double A = 0.0; for (int i = 0; i < this->Legs.size(); ++i) A += this->Legs[i].x * this->Legs[i].x;
128 double B = 0.0; for (int i = 0; i < this->Legs.size(); ++i) B += this->Legs[i].y * this->Legs[i].y;
129 double C = 0.0; for (int i = 0; i < this->Legs.size(); ++i) C += this->Legs[i].x * this->Legs[i].y;
130 double D = 0.0; for (int i = 0; i < this->Legs.size(); ++i) D += this->Legs[i].x * this->Legs[i].z;
131 double E = 0.0; for (int i = 0; i < this->Legs.size(); ++i) E += this->Legs[i].y * this->Legs[i].z;
132 double F = 0.0; for (int i = 0; i < this->Legs.size(); ++i) F += this->Legs[i].x;
133 double G = 0.0; for (int i = 0; i < this->Legs.size(); ++i) G += this->Legs[i].y;
134 double H = 0.0; for (int i = 0; i < this->Legs.size(); ++i) H += this->Legs[i].z;
135
136 Array x(3); x[0] = D, x[1] = E, x[2] = H;
137 Array ta(3); ta[0] = A, ta[1] = C, ta[2] = F;
138 Array tb(3); tb[0] = C, tb[1] = B, tb[2] = G;
139 Array tc(3); tc[0] = F, tc[1] = G, tc[2] = double(this->Legs.size());
140 Matrix AA(3, Array(3)); AA[0] = ta; AA[1] = tb; AA[2] = tc;
141 Array bb = Givens_rotation(AA, x);
142 a = bb[0]; b = bb[1]; c = bb[2];
143 double d = (-a * Legs[0].x - b * Legs[0].y - c * Legs[0].z);
144 std::cerr << a << " " << b << " " << c << " " << d << std::endl;
145 }
146
147 // 平面の　方程式をax+by+cz + d = 0と表現する
148 POINT G = this->G_COM;
149 POINT Gc; Gc.x = a * G.x, Gc.y = b * G.y, Gc.z = c * G.z; // 嘘
150 double theta = atan(a / b); //rad
151 double xi = theta;
152 POINT P;
153 for (int i = 0; i < this->Legs.size(); ++i) {
154 // 隣り合う2脚が回転軸になる
155 int j = (i == this->Legs.size() - 1 ? 0 : i + 1);
156
157 // P : (x - x_0) + (y - y_0) + ( z - z_0) + t(ls_vec) において二分探索してPを見つける
158 POINT ls_vec = this->Legs[i] - this->Legs[j];
159 double MAX_t = 100000000;
160 double MIN_t = -100000000;
161 double t = (MAX_t + MIN_t) / 3.0;
162 {
163 double d = (MAX_t - MIN_t) / 3.0;
164 while (abs(MAX_t - MIN_t) < 0.0001) {
165 d = (MAX_t - MIN_t) / 2.0;
166 double sa = Dot(ls_vec, G_COM - (Legs[i] + (MIN_t + d) * ls_vec));
167 double sb = Dot(ls_vec, G_COM - (Legs[i] + (MIN_t + 2 * d) * ls_vec));
168 sa /= Mag(ls_vec) * Mag(G_COM - (Legs[i] + (MIN_t + d) * ls_vec));
169 sb /= Mag(ls_vec) * Mag(G_COM - (Legs[i] + (MIN_t + 2 * d) * ls_vec));
170 //std::cerr << sa << " " << sb << std::endl;
171 if (abs(sa) > abs(sb)) MIN_t = t;
172 else MAX_t = t + d;
173 }
174 P = Legs[i] + d * ls_vec;
175 }
176 double h = Mag(G - P);
177 double d = Mag(Gc - P);
178 std::cerr << h << " " << d << std::endl;
179 std::cerr << "!" << sqrt(h * h + d * d) - (h * cos(theta) + d * sin(theta)) << std::endl;
180 ret = (std::min)(ret, sqrt(h * h + d * d) - (h * cos(theta) + d * sin(theta)));
181 }
182 std::cerr << std::fixed << "Now S_NE Score = " << ret << std::endl;
183 return ret;
184 }
