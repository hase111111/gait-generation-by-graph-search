49template <typename T>
50std::string EnumToStringRemoveTopK(const T& enum_value)
51{
52 // 型チェック Tが enum型であることをチェックする．
53 static_assert(std::is_enum<T>::value, "引数は enum，あるいは enum classである必要があります．");
54
55 std::string str = static_cast<std::string>(magic_enum::enum_name(enum_value));
56
57 if (str.size() > 0 && str[0] == 'k')
58 {
59 // 先頭のkを削除する．
60 str.erase(0, 1);
61 }
62
63 return str;
64}

128template <typename T>
129std::string GetTypeName(const T& type)
130{
131 std::string str = typeid(type).name();
132
133 std::vector<std::string> eliminate
134 {
135 "class ", "struct ", "designlab::", ",void"
136 };
137
138 for (const auto& e : eliminate)
139 {
140 while (true)
141 {
142 auto pos = str.find(e);
143
144 if (pos == std::string::npos)
145 {

149 str.erase(pos, e.size());

151 }
