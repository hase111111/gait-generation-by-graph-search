923void detectLandingPoint::correctDisparity(double* X, double* Y, double* Z, double theta)
924{
925 double Xc, Yc, Zc;
926 double disp, trueDisp;
927 double x, y;
928 double Xct, Yct, Zct;
929 double Xt, Yt, Zt;
930 Xc = *Y - _cameraPosi.y;
931 Yc = -(*X) * std::sin(theta) + ((*Z) - _cameraPosi.z) * std::cos(theta);
932 Zc = (*X) * std::cos(theta) + ((*Z) - _cameraPosi.z) * std::sin(theta);
933 disp = _focalLength * BASE_LINE / Zc;
934 x = Xc * disp / BASE_LINE - _priPoint.x;
935 y = Yc * disp / BASE_LINE - _priPoint.y;
936 trueDisp = ((y + _priPoint.y) * std::cos(theta) + _focalLength * std::sin(theta)) * BASE_LINE / (BODYHEIGHT - _cameraPosi.z);
937 Xct = (x + _priPoint.x) * BASE_LINE / trueDisp;
938 Yct = (y + _priPoint.y) * BASE_LINE / trueDisp;
939 Zct = _focalLength * BASE_LINE / trueDisp;
940 Xt = -Yct * std::sin(theta) + Zct * std::cos(theta);
941 Yt = Xct + _cameraPosi.y;
942 Zt = BODYHEIGHT;
943 * X = Xt;
944 * Y = Yt;
945 * Z = Zt;
946}
